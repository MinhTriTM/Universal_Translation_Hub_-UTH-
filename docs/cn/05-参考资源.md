### 1.3 MiMo API 调用示例

```python
# MiMo-V2.5 文本翻译示例
import httpx

async def translate_text(source_text: str, source_lang: str) -> str:
    """使用 MiMo-V2.5 翻译文本"""
    async with httpx.AsyncClient() as client:
        response = await client.post(
            "https://api.mimo.ai/v1/chat/completions",
            headers={
                "Authorization": f"Bearer {MIMO_API_KEY}",
                "Content-Type": "application/json"
            },
            json={
                "model": "mimo-v2.5",
                "messages": [
                    {
                        "role": "system",
                        "content": "你是专业翻译，将以下文本翻译成越南语。保留原文格式和术语。"
                    },
                    {
                        "role": "user",
                        "content": f"将以下{source_lang}文本翻译成越南语：\n\n{source_text}"
                    }
                ],
                "temperature": 0.3,
                "max_tokens": 4096
            }
        )
        return response.json()["choices"][0]["message"]["content"]
```

```python
# MiMo-V2.5-TTS 语音合成示例
async def text_to_speech(text: str, voice_id: str = "default") -> bytes:
    """使用 MiMo-TTS 合成语音"""
    async with httpx.AsyncClient() as client:
        response = await client.post(
            "https://api.mimo.ai/v1/audio/speech",
            headers={
                "Authorization": f"Bearer {MIMO_API_KEY}",
                "Content-Type": "application/json"
            },
            json={
                "model": "mimo-v2.5-tts",
                "input": text,
                "voice": voice_id,
                "language": "vi",
                "speed": 1.0
            }
        )
        return response.content  # WAV 音频数据
```

```python
# MiMo-TTS-VoiceClone 声音克隆示例
async def clone_voice(
    sample_audio_path: str,
    character_name: str
) -> dict:
    """使用 VoiceClone 克隆声音"""
    async with httpx.AsyncClient() as client:
        # 上传参考音频
        with open(sample_audio_path, "rb") as f:
            audio_data = f.read()

        response = await client.post(
            "https://api.mimo.ai/v1/audio/voiceclone",
            headers={
                "Authorization": f"Bearer {MIMO_API_KEY}",
            },
            files={"audio": ("sample.wav", audio_data, "audio/wav")},
            data={
                "model": "mimo-tts-voiceclone",
                "name": character_name,
                "language": "vi"
            }
        )
        return response.json()  # 声音配置 ID
```

**常用 FFmpeg 命令：**

```bash
# 分离音轨
ffmpeg -i input.mp4 -vn -acodec pcm_s16le audio.wav

# 分离视频轨（无音频）
ffmpeg -i input.mp4 -an -c copy video_only.mp4

# 混合音频和视频
ffmpeg -i video_only.mp4 -i tts_audio.wav -c:v copy -c:a aac output.mp4

# 提取字幕
ffmpeg -i input.mkv -map 0:s:0 subtitles.srt

# 烧录字幕
ffmpeg -i input.mp4 -vf "subtitles=subtitles.srt" output.mp4

# 调整音频音量
ffmpeg -i audio.wav -af "volume=0.5" output.wav

# 混合多个音轨
ffmpeg -i background.wav -i voice.wav -filter_complex "amix=inputs=2:duration=longest" mixed.wav
```
